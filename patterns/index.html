Circles & Underlines

Hand-drawn emphasis elements for words and data points. CSS-mask technique — any color. Weight: 90

EMPHASIS ONLY, NOT DECORATIVE — Every circle/underline must wrap a specific word or data point. White PNG masks + backgroundColor = any color. Never use hue-rotate or tinted PNGs. 90
How it works — White PNG used as CSS mask-image. A colored div behind the text has its shape cut by the mask. Change backgroundColor to change the color. Works in any context: static HTML, social post images, presentations, web pages.
STATIC BY DEFAULT — For social post images, email graphics, one-pagers, and any static/image output: just use the mask technique with no animation. Animation is optional and only for interactive contexts (presentations, web pages, app UIs). Most brand assets are rendered as static images. 75

Copy: Animated reveal (conic sweep for circles, linear sweep for underlines)
JavaScript 
/* Animated Emphasis Reveal
   Circle: conic-gradient sweep from 12 o'clock with noisy fade edge
   Underline: left-to-right linear-gradient with noisy fade edge
   Both use requestAnimationFrame + easeInOutCubic
*/
const FADE_ZONE = 0.15, NOISE_STRENGTH = 0.7, NOISE_STOPS = 40;

function easeInOutCubic(t) {
  return t < 0.5 ? 4*t*t*t : 1 - Math.pow(-2*t + 2, 3) / 2;
}

// Circle: conic sweep from top
function circleRevealGradient(progress) {
  const sweepDeg = progress * 360, fadeDeg = FADE_ZONE * 360;
  const solidEnd = Math.max(0, sweepDeg - fadeDeg);
  const stops = [`rgba(0,0,0,1) 0deg`, `rgba(0,0,0,1) ${solidEnd.toFixed(1)}deg`];
  for (let i = 0; i <= NOISE_STOPS; i++) {
    const t = i / NOISE_STOPS, deg = solidEnd + t * fadeDeg;
    if (deg > sweepDeg) break;
    const baseAlpha = 1 - t;
    const noise = (Math.random() - 0.5) * 2 * NOISE_STRENGTH;
    const alpha = Math.max(0, Math.min(1, baseAlpha + noise * (1 - Math.abs(baseAlpha - 0.5) * 2)));
    stops.push(`rgba(0,0,0,${alpha.toFixed(3)}) ${deg.toFixed(1)}deg`);
  }
  stops.push(`rgba(0,0,0,0) ${sweepDeg.toFixed(1)}deg`, `rgba(0,0,0,0) 360deg`);
  return `conic-gradient(from -90deg at 50% 50%, ${stops.join(', ')})`;
}

// Underline: left-to-right linear sweep
function underlineRevealGradient(progress) {
  const sweepPct = progress * 100, fadePct = FADE_ZONE * 100;
  const solidEnd = Math.max(0, sweepPct - fadePct);
  const stops = [`rgba(0,0,0,1) 0%`, `rgba(0,0,0,1) ${solidEnd.toFixed(1)}%`];
  for (let i = 0; i <= NOISE_STOPS; i++) {
    const t = i / NOISE_STOPS, pct = solidEnd + t * fadePct;
    if (pct > sweepPct) break;
    const baseAlpha = 1 - t;
    const noise = (Math.random() - 0.5) * 2 * NOISE_STRENGTH;
    const alpha = Math.max(0, Math.min(1, baseAlpha + noise * (1 - Math.abs(baseAlpha - 0.5) * 2)));
    stops.push(`rgba(0,0,0,${alpha.toFixed(3)}) ${pct.toFixed(1)}%`);
  }
  stops.push(`rgba(0,0,0,0) ${sweepPct.toFixed(1)}%`, `rgba(0,0,0,0) 100%`);
  return `linear-gradient(to right, ${stops.join(', ')})`;
}

function animateReveal(el, gradientFn, duration) {
  el.style.opacity = '1';
  let start = null;
  function frame(now) {
    if (!start) start = now;
    const raw = Math.min((now - start) / duration, 1);
    const grad = gradientFn(easeInOutCubic(raw));
    el.style.webkitMaskImage = grad;
    el.style.maskImage = grad;
    el.style.webkitMaskSize = '100% 100%';
    el.style.maskSize = '100% 100%';
    if (raw < 1) requestAnimationFrame(frame);
    else { el.style.webkitMaskImage = ''; el.style.maskImage = ''; }
  }
  requestAnimationFrame(frame);
}
