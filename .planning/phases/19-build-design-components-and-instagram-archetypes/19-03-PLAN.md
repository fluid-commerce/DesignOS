```typescript
import { test, expect } from '@playwright/test';
import * as fs from 'node:fs';
import * as path from 'node:path';

// All 6 archetype slugs
const ARCHETYPE_SLUGS = [
  'hero-stat',
  'photo-bg-overlay',
  'split-photo-text',
  'quote-testimonial',
  'minimal-statement',
  'data-dashboard',
];

const ARCHETYPES_DIR = path.resolve(__dirname, '../../archetypes');

// Helper: read schema.json for an archetype
function loadSchema(slug: string) {
  const schemaPath = path.join(ARCHETYPES_DIR, slug, 'schema.json');
  return JSON.parse(fs.readFileSync(schemaPath, 'utf-8'));
}

// Helper: create a test iteration with an archetype's schema via API
async function createTestIteration(request: any, slug: string, schema: any) {
  // 1. Create campaign
  const campRes = await request.post('/api/campaigns', {
    data: { name: `E2E-archetype-${slug}-${Date.now()}` }
  });
  const camp = await campRes.json();

  // 2. Create creation
  const createRes = await request.post('/api/creations', {
    data: {
      campaignId: camp.id,
      name: `${slug}-test`,
      contentType: 'social',
      channel: 'instagram',
      dimensions: { width: schema.width, height: schema.height }
    }
  });
  const creation = await createRes.json();

  // 3. Create slide
  const slideRes = await request.post('/api/slides', {
    data: { creationId: creation.id }
  });
  const slide = await slideRes.json();

  // 4. Create iteration with schema as slot_schema
  // Read the archetype HTML to use as iteration content
  const htmlPath = path.join(ARCHETYPES_DIR, slug, 'index.html');
  const htmlContent = fs.readFileSync(htmlPath, 'utf-8');

  const iterRes = await request.post('/api/iterations', {
    data: {
      slideId: slide.id,
      htmlPath: `archetypes/${slug}/index.html`,
      slotSchema: JSON.stringify(schema),
      html: htmlContent
    }
  });
  const iteration = await iterRes.json();

  return { campaignId: camp.id, creationId: creation.id, slideId: slide.id, iterationId: iteration.id };
}
```
