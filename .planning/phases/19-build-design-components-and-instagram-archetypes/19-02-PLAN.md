```typescript
export interface SlotSchema {
  templateId?: string;    // OMIT for archetypes
  width: number;          // 1080
  height: number;         // 1080
  fields: SlotField[];
  brush?: string | null;  // null for archetypes
  brushLabel?: string;

  carouselCount?: number; // omit for single-frame
}

```
