       // Build iteration context from current session state
       const currentRound = activeSessionData.lineage.rounds
         ? Math.max(...activeSessionData.lineage.rounds.map(r => r.roundNumber), 0)
         : 0;
       const originalPrompt = activeSessionData.lineage.rounds?.[0]?.prompt ?? '';

       const iterCtx = buildIterationContext({
         variations: activeSessionData.variations,
         annotations,
         statuses,
         currentRound,
         originalPrompt,
       });

       generate(text, {
         skillType: 'social',
         sessionId: activeSessionId!,
         iterationContext: iterCtx,
       });
     } else {
       generate(text, { skillType: 'social' });
     }
     setPrompt('');
   };
